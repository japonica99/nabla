155 void EmitterState::EndedGroup(GroupType::value type) {
156  if (m_groups.empty()) {
157  if (type == GroupType::Seq) {
158  return SetError(ErrorMsg::UNEXPECTED_END_SEQ);
159  } else {
160  return SetError(ErrorMsg::UNEXPECTED_END_MAP);
161  }
162  }
163 
164  // get rid of the current group
165  {
166  std::unique_ptr<Group> pFinishedGroup = std::move(m_groups.back());
167  m_groups.pop_back();
168  if (pFinishedGroup->type != type) {
169  return SetError(ErrorMsg::UNMATCHED_GROUP_TAG);
170  }
171  }
172 
173  // reset old settings
174  std::size_t lastIndent = (m_groups.empty() ? 0 : m_groups.back()->indent);
175  assert(m_curIndent >= lastIndent);
176  m_curIndent -= lastIndent;
177 
178  // some global settings that we changed may have been overridden
179  // by a local setting we just popped, so we need to restore them
180  m_globalModifiedSettings.restore();
181 
182  ClearModifiedSettings();
183 }
