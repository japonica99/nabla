333 void Stream::StreamInUtf16() const {
334  unsigned long ch = 0;
335  unsigned char bytes[2];
336  int nBigEnd = (m_charSet == utf16be) ? 0 : 1;
337 
338  bytes[0] = GetNextByte();
339  bytes[1] = GetNextByte();
340  if (!m_input.good()) {
341  return;
342  }
343  ch = (static_cast<unsigned long>(bytes[nBigEnd]) << 8) |
344  static_cast<unsigned long>(bytes[1 ^ nBigEnd]);
345 
346  if (ch >= 0xDC00 && ch < 0xE000) {
347  // Trailing (low) surrogate...ugh, wrong order
348  QueueUnicodeCodepoint(m_readahead, CP_REPLACEMENT_CHARACTER);
349  return;
350  } else if (ch >= 0xD800 && ch < 0xDC00) {
351  // ch is a leading (high) surrogate
352 
353  // Four byte UTF-8 code point
354 
355  // Read the trailing (low) surrogate
356  for (;;) {
357  bytes[0] = GetNextByte();
358  bytes[1] = GetNextByte();
359  if (!m_input.good()) {
360  QueueUnicodeCodepoint(m_readahead, CP_REPLACEMENT_CHARACTER);
361  return;
362  }
363  unsigned long chLow = (static_cast<unsigned long>(bytes[nBigEnd]) << 8) |
364  static_cast<unsigned long>(bytes[1 ^ nBigEnd]);
365  if (chLow < 0xDC00 || chLow >= 0xE000) {
366  // Trouble...not a low surrogate. Dump a REPLACEMENT CHARACTER into the
367  // stream.
368  QueueUnicodeCodepoint(m_readahead, CP_REPLACEMENT_CHARACTER);
369 
370  // Deal with the next UTF-16 unit
371  if (chLow < 0xD800 || chLow >= 0xE000) {
372  // Easiest case: queue the codepoint and return
373  QueueUnicodeCodepoint(m_readahead, ch);
374  return;
375  } else {
376  // Start the loop over with the new high surrogate
377  ch = chLow;
378  continue;
379  }
380  }
381 
382  // Select the payload bits from the high surrogate
383  ch &= 0x3FF;
384  ch <<= 10;
385 
386  // Include bits from low surrogate
387  ch |= (chLow & 0x3FF);
388 
389  // Add the surrogacy offset
390  ch += 0x10000;
391  break;
392  }
393  }
394 
395  QueueUnicodeCodepoint(m_readahead, ch);
396 }
